        #word { font-size: 2.5em; color: #333; margin-bottom: 20px; cursor: pointer; }

        #definition, #example { font-size: 1.2em; color: #666; min-height: 50px; }

<div class="card">

    <h2>IELTS Band 8.0 Vocabulary</h2>

    <div id="word">Click to Reveal</div>

    <div id="definition" style="display:none;"></div>

    <div id="example" style="display:none;"></div>

    <button onclick="nextCard()">Next Word</button>

</div>


<script>

    // --- STEP 3: PASTE YOUR PUBLISHED SHEET URL HERE ---

    const GOOGLE_SHEET_URL = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vQbYZwC-HcUDJK0JVkW4SC8j0DLDtVnCsR8uTmE_z3iSS8H9i9lmt5jB8FMVSVjQ_SDj_N56H-PfCtz/pub?gid=0&single=true&output=csv';

    // ----------------------------------------------------


    let vocabData = [];

    let currentIndex = -1;


    // Function to load the CSV data from Google Sheets

    async function loadVocab() {

        try {

            const response = await fetch(GOOGLE_SHEET_URL);

            const csvText = await response.text();

            

            // Basic CSV parsing: split by line, then by comma (skip the header row)

            const rows = csvText.split('\n').slice(1); 

            

            vocabData = rows.map(row => {

                const columns = row.split(',');

                if (columns.length >= 4) {

                    return {

                        word: columns[0].trim(),

                        definition: columns[1].trim(),

                        example: columns[2].trim(),

                        band: columns[3].trim()

                    };

                }

                return null;

            }).filter(Boolean); // Filter out any empty/bad rows


            if (vocabData.length > 0) {

                vocabData = vocabData.sort(() => Math.random() - 0.5); // Shuffle the array

                nextCard();

            } else {

                document.getElementById('word').innerText = 'Error: No data loaded.';

            }


        } catch (error) {

            console.error('Error loading vocabulary:', error);

            document.getElementById('word').innerText = 'Error loading data. Check console.';

        }

    }


    // Function to show the next card

    function nextCard() {

        currentIndex = (currentIndex + 1) % vocabData.length;

        const currentWord = vocabData[currentIndex];


        const wordElement = document.getElementById('word');

        const defElement = document.getElementById('definition');

        const exElement = document.getElementById('example');


        // Reset and show word only

        wordElement.innerText = currentWord.word;

        defElement.innerText = currentWord.definition;

        exElement.innerText = currentWord.example;


        defElement.style.display = 'none';

        exElement.style.display = 'none';


        wordElement.onclick = () => {

            if (defElement.style.display === 'none') {

                defElement.style.display = 'block';

                exElement.style.display = 'block';

            } else {

                defElement.style.display = 'none';

                exElement.style.display = 'none';

            }

        };

    }


    // Load data when the page opens

    loadVocab();

</script>
